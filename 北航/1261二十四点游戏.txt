bool calcu(double n1,char p,double n2,double &result)
{
    if(p=='/'&&n2==0)
        return false;
    else
    {
        switch(p)
        {
            case '+':result=n1+n2;return true;
            case '-':result=n1-n2;return true;
            case '*':result=n1*n2;return true;
            case '/':result=n1/n2;return true;
            default:return false;
        }
    }
}
