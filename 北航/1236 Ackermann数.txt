#include<iostream>
#include<algorithm>
#include<sstream>
#include<vector>
#include<iomanip>
#include<cmath>
using namespace std;
int A(int m,int n)
{
	if(m==0) return n+1;
	if(m>0 && n==0) return A(m-1,1);
	return A(m-1,A(m,n-1));
}
int main()
{
	int n;
	cin>>n;
	while(n--)
	{
		int a,b;
		cin>>a>>b;
		cout<<A(a,b)<<endl;
	}
	return 0;
}
